// キーボードナビゲーション対応 document.addEventListener('keydown', function(event) { // Tab キーでのフォーカス管理 if (event.key === 'Tab') { // デフォルトの Tab 動作を維持 return; } // 矢印キーでのゲーム切り替え if (event.key === 'ArrowLeft' || event.key === 'ArrowRight') { const activeButton = document.querySelector('.game-nav-button.active'); if (activeButton) { event.preventDefault(); const buttons = Array.from(document.querySelectorAll('.game-nav-button')); const currentIndex = buttons.indexOf(activeButton); let nextIndex; if (event.key === 'ArrowRight') { nextIndex = (currentIndex + 1) % buttons.length; } else { nextIndex = (currentIndex - 1 + buttons.length) % buttons.length; } buttons[nextIndex].click(); buttons[nextIndex].focus(); } } // Enter または Space キーでボタン実行 if (event.key === 'Enter' || event.key === ' ') { const focusedElement = document.activeElement; if (focusedElement && focusedElement.classList.contains('game-nav-button')) { event.preventDefault(); focusedElement.click(); } } }); // フォーカス管理の改善 document.querySelectorAll('.game-nav-button').forEach(button => { button.addEventListener('focus', function() { // フォーカス時にボタンを強調表示 this.style.outline = '4px solid #4A90E2'; this.style.outlineOffset = '2px'; }); button.addEventListener('blur', function() { // フォーカス解除時にアウトラインを削除 this.style.outline = ''; this.style.outlineOffset = ''; }); });
